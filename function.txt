#include <iostream>
#include <cmath> 

using namespace std;

bool i;
int c=0;
float a=2;
float del=0.6;
float x;
float y;
float b[6] = {1.16, 1.32, 1.47, 1.65, 1.93};
    
int main()
{
    cout << "Запустить задачу A/B? (1/0)" << endl;
    cin >> i;
    if (i)
    {
    	for (x = 1.2; x<4.3; x=x+del) 
    	{
        y=pow((log10(a+x)),2)/pow((a+x),2);
        b[c]=y;
        c=c+1;
    	}
    }
    else
    {
    	for (int c = 0; c<5; c++) 
    	{
        y=pow((log10(a+b[c])),2)/pow((a+b[c]),2);
        b[c]=y;
    	}
    }
    for (c = 0; c<6; c++)
    {
      if (b[c]!=0)
      {
        cout << b[c] << endl;
      }
    }
}
